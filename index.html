-- ALKAM MALİ - ACİL DÜZELTME -- Amaç: -- 1) cari_fee_history şemasını uygulamanın beklediği alanlarla uyumlu hale getirmek -- 2) fee_amount / note ile monthly_fee_amount / reason_note alanlarını senkronlamak -- 3) v_cari_fee_timeline ve v_monthly_fee_accrual_control görünümlerini yeniden kurmak -- 4) siteyi tekrar ayağa kaldırmak begin; -- 0) Tablo yoksa sadece uygulamanın beklediği minimal uyumlu şemayı kur create table if not exists public.cari_fee_history ( id uuid primary key default gen_random_uuid(), cari_id uuid not null references public.cari_cards(id) on delete cascade, valid_from date not null, valid_to date, fee_amount numeric(18,2), monthly_fee_amount numeric(18,2), note text, reason_note text, currency_code text default 'TRY', created_by text default 'system', is_active boolean not null default true, created_at timestamptz not null default now(), updated_at timestamptz not null default now() ); -- 1) Eksik kolonları ekle alter table public.cari_fee_history add column if not exists fee_amount numeric(18,2); alter table public.cari_fee_history add column if not exists monthly_fee_amount numeric(18,2); alter table public.cari_fee_history add column if not exists note text; alter table public.cari_fee_history add column if not exists reason_note text; alter table public.cari_fee_history add column if not exists currency_code text default 'TRY'; alter table public.cari_fee_history add column if not exists created_by text default 'system'; alter table public.cari_fee_history add column if not exists is_active boolean not null default true; alter table public.cari_fee_history add column if not exists created_at timestamptz not null default now(); alter table public.cari_fee_history add column if not exists updated_at timestamptz not null default now(); -- 2) Çift yönlü senkron update public.cari_fee_history set fee_amount = coalesce(fee_amount, monthly_fee_amount), monthly_fee_amount = coalesce(monthly_fee_amount, fee_amount), note = coalesce(note, reason_note), reason_note = coalesce(reason_note, note), currency_code = coalesce(currency_code, 'TRY'), created_by = coalesce(created_by, 'system'), is_active = coalesce(is_active, true), updated_at = now(); create index if not exists idx_cari_fee_history_cari on public.cari_fee_history(cari_id); create index if not exists idx_cari_fee_history_period on public.cari_fee_history(cari_id, valid_from, valid_to); -- 3) Uygulama uyumlu timeline görünümü create or replace view public.v_cari_fee_timeline as select f.id, f.cari_id, c.cari_name, c.cari_code, f.valid_from, f.valid_to, coalesce(f.fee_amount, f.monthly_fee_amount, 0)::numeric(18,2) as fee_amount, coalesce(f.monthly_fee_amount, f.fee_amount, 0)::numeric(18,2) as current_fee, coalesce(f.fee_amount, f.monthly_fee_amount, 0)::numeric(18,2) as amount, coalesce(f.note, f.reason_note, '') as note, coalesce(f.reason_note, f.note, '') as change_reason, f.currency_code, f.created_by, coalesce(f.is_active, true) as is_active, f.created_at, f.updated_at from public.cari_fee_history f left join public.cari_cards c on c.id = f.cari_id; -- 4) Ücret kontrol görünümü create or replace view public.v_monthly_fee_accrual_control as with fee_months as ( select f.cari_id, c.cari_name, c.cari_code, gs::date as period_start, (date_trunc('month', gs)::date + interval '1 month - 1 day')::date as period_end, coalesce(f.fee_amount, f.monthly_fee_amount, 0)::numeric(18,2) as expected_amount from public.cari_fee_history f join public.cari_cards c on c.id = f.cari_id cross join lateral generate_series( date_trunc('month', greatest(f.valid_from, date '2026-01-01'))::date, date_trunc('month', coalesce(f.valid_to, current_date))::date, interval '1 month' ) gs where coalesce(f.is_active, true) = true and coalesce(f.fee_amount, f.monthly_fee_amount, 0) > 0 and coalesce(f.valid_to, current_date) >= date '2026-01-01' ), existing as ( select l.cari_id, date_trunc('month', l.line_date)::date as period_start, count(*) as tahakkuk_satir_sayisi, sum(coalesce(l.debit,0))::numeric(18,2) as existing_amount, string_agg(coalesce(l.description,''), ' || ' order by l.line_date, l.id) as note from public.cari_ekstre_lines l where l.movement_type = 'tahakkuk' and l.line_date >= date '2026-01-01' group by l.cari_id, date_trunc('month', l.line_date)::date ) select m.cari_id, m.cari_name, m.cari_code, m.period_start, m.period_end, m.expected_amount, m.expected_amount as expected_fee_amount, m.expected_amount as current_fee, coalesce(e.existing_amount,0)::numeric(18,2) as existing_amount, coalesce(e.existing_amount,0)::numeric(18,2) as actual_amount, coalesce(e.existing_amount,0)::numeric(18,2) as posted_amount, (coalesce(e.existing_amount,0) - m.expected_amount)::numeric(18,2) as diff_amount, case when e.cari_id is null then 'eksik' when e.tahakkuk_satir_sayisi > 1 then 'muhtemel_duplicate' when coalesce(e.existing_amount,0) <> m.expected_amount then 'farkli_tutar' else 'tamam' end as control_status, coalesce(e.note, '') as note, case when e.cari_id is null then 'Beklenen ay tahakkuku bulunamadı' when e.tahakkuk_satir_sayisi > 1 then 'Aynı ayda birden fazla tahakkuk var' when coalesce(e.existing_amount,0) <> m.expected_amount then 'Beklenen ücret ile bulunan tahakkuk farklı' else 'Kontrol tamam' end as issue_reason, case when e.cari_id is null then 'Eksik tahakkuk' when e.tahakkuk_satir_sayisi > 1 then 'Muhtemel çift kayıt' when coalesce(e.existing_amount,0) <> m.expected_amount then 'Farklı tutar' else 'Tamam' end as status_note from fee_months m left join existing e on e.cari_id = m.cari_id and e.period_start = m.period_start; create or replace view public.v_missing_monthly_fee_accruals as select * from public.v_monthly_fee_accrual_control where control_status in ('eksik','farkli_tutar','muhtemel_duplicate'); commit; -- 5) Kontrol çıktıları select 'cari_fee_history_kolon_kontrolu' as kontrol, column_name from information_schema.columns where table_schema='public' and table_name='cari_fee_history' and column_name in ('fee_amount','monthly_fee_amount','note','reason_note') order by column_name; select 'timeline_ornek' as kontrol, * from public.v_cari_fee_timeline limit 5; select 'fee_control_ornek' as kontrol, * from public.v_monthly_fee_accrual_control limit 10;
